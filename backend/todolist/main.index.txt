<!DOCTYPE html>
<html>
<body>

<h2>Enter Name</h2>

<input id="name" placeholder="Enter name">
<button onclick="sendData()">Submit</button>

<script>
function sendData() {
    let name = document.getElementById("name").value;

    fetch("http://127.0.0.1:8000/submit", {
        method: "POST",
        headers: {
            "Content-Type": "application/json"
        },
        body: JSON.stringify({
            name: name
        })
    })
    .then(res => res.json())
    .then(data => {
        alert(data.msg);
    });
}
</script>

</body>
